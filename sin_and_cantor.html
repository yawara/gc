from math import modf

def cantor(x, N=100):
    if x > 1 or x < 0:
        raise Exception
    if x == 0:
        return 0
    if x == 1:
        return 1

    rtv = 0

    tmp = x

            rtv += 2**(-i)
            break

    return rtv


def np_cantor(X):
    Y = np.zeros(len(X))
    for i, x in enumerate(X):
        Y[i] = cantor(x)

    return Y
